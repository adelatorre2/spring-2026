\makeatletter
\newcommand{\problem}[2][]{%
  \def\prob@title{Problem #2\if\relax\detokenize{#1}\relax\else\ (\textit{#1})\fi}%
  \phantomsection%
  \pdfbookmark[1]{\prob@title}{prob#2}%
  \addcontentsline{toc}{section}{\prob@title}%
  \section*{\prob@title}%
  \vspace{-0.25em}\hrule\vspace{0.8em}%
}
\makeatother
