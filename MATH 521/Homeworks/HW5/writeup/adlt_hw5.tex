\problem[]{1}
Let $\{a_n\}_{n\ge 1}$ be a sequence of positive real numbers. Prove that
$\lim_{n\to\infty} a_n = +\infty$ if and only if
$\lim_{n\to\infty} \frac{1}{a_n} = 0$.

\begin{scratch}
This is an ``if and only if'' statement, so we prove both directions using the
definitions.

Because $a_n>0$ for all $n$, the reciprocal $\frac{1}{a_n}$ is well-defined and
positive. The key observation is that for positive numbers, taking reciprocals
reverses inequalities: for $x>0$ and $M>0$,
\[
x > M \quad \Longleftrightarrow \quad \frac{1}{x} < \frac{1}{M}.
\]
So ``$a_n$ eventually exceeds every $M$'' is the same as
``$\frac{1}{a_n}$ eventually falls below every $\varepsilon$'' (after choosing
$\varepsilon = 1/M$ or $M=1/\varepsilon$).
\end{scratch}

\textbf{Proof.}\\
($\Rightarrow$) Assume $\lim_{n\to\infty} a_n = +\infty$. By definition, for every
$M>0$ there exists $N\in\mathbb{N}$ such that $a_n > M$ for all $n\ge N$.
Let $\varepsilon>0$ be given and choose $M = 1/\varepsilon$ (note $M>0$).
Then there exists $N$ such that for all $n\ge N$,
\[
a_n > \frac{1}{\varepsilon}.
\]
Since $a_n>0$, taking reciprocals yields
\[
\frac{1}{a_n} < \varepsilon \qquad \text{for all } n\ge N.
\]
Thus for all $n\ge N$ we have $\left|\frac{1}{a_n}-0\right|=\frac{1}{a_n}<\varepsilon$,
which shows $\lim_{n\to\infty}\frac{1}{a_n}=0$.

($\Leftarrow$) Assume $\lim_{n\to\infty}\frac{1}{a_n}=0$. By definition, for every
$\varepsilon>0$ there exists $N\in\mathbb{N}$ such that for all $n\ge N$,
\[
\left|\frac{1}{a_n}-0\right|<\varepsilon.
\]
Because $\frac{1}{a_n}>0$, this is the same as $\frac{1}{a_n}<\varepsilon$.
Now let $M>0$ be given and choose $\varepsilon = 1/M$ (note $\varepsilon>0$).
Then there exists $N$ such that for all $n\ge N$,
\[
\frac{1}{a_n} < \frac{1}{M}.
\]
Since $a_n>0$, taking reciprocals reverses the inequality and gives
\[
a_n > M \qquad \text{for all } n\ge N.
\]
This is exactly the definition of $\lim_{n\to\infty} a_n = +\infty$.
\qed

\problem[]{2}
Let $\{a_n\}_{n\ge 1}$ be a sequence. Prove that if $\{a_{k_n}\}_{n\ge 1}$ is a
subsequence of $\{a_n\}_{n\ge 1}$ that converges to $L\in\mathbb{R}$, then
\[
\liminf_{n\to\infty} a_n \le L \le \limsup_{n\to\infty} a_n.
\]
(Hint: As $\{a_n\}_{n\ge 1}$ is not necessarily bounded, we have to consider
the cases $\liminf_{n\to\infty} a_n = -\infty$ and
$\limsup_{n\to\infty} a_n = +\infty$. However, the proof is much easier in
these cases.)

\begin{scratch}
The core idea is: compare each subsequence term $a_{k_n}$ to the \emph{tail} bounds of the original sequence.
Fix a tail starting at $N$. Then
\[
u_N := \inf\{a_m:m\ge N\}\quad\text{and}\quad v_N:=\sup\{a_m:m\ge N\}
\]
satisfy $u_N \le a_m \le v_N$ for every $m\ge N$.

Now use the fact that $k_n\to\infty$ (because $\{k_n\}$ is strictly increasing), so for any fixed $N$,
eventually $k_n\ge N$. Hence for all large $n$,
\[
u_N \le a_{k_n} \le v_N.
\]
Taking $n\to\infty$ gives $u_N \le L \le v_N$ for each $N$.
Finally, let $N\to\infty$: by definition,
\[
\liminf_{n\to\infty} a_n=\lim_{N\to\infty}u_N,\qquad
\limsup_{n\to\infty} a_n=\lim_{N\to\infty}v_N,
\]
and passing to the limit preserves the inequalities.
(If $\liminf=-\infty$ or $\limsup=+\infty$, the corresponding inequality is automatic.)
\end{scratch}

\textbf{Proof.}\\
Let $\{a_{k_n}\}_{n\ge 1}$ be a subsequence of $\{a_n\}_{n\ge 1}$ and suppose
\[
\lim_{n\to\infty} a_{k_n}=L\in\mathbb{R}.
\]

\emph{Case 1:} $\liminf_{n\to\infty} a_n=-\infty$.  
Then $\liminf_{n\to\infty} a_n \le L$ holds automatically, since $-\infty\le L$.

\emph{Case 2:} $\limsup_{n\to\infty} a_n=+\infty$.  
Then $L\le \limsup_{n\to\infty} a_n$ holds automatically, since $L\le +\infty$.

So assume from now on that $\{a_n\}$ is bounded (equivalently, both $\liminf a_n$ and $\limsup a_n$
are finite real numbers). For each $N\in\mathbb{N}$ define the tail infimum and tail supremum
\[
u_N=\inf\{a_m:m\ge N\},\qquad v_N=\sup\{a_m:m\ge N\}.
\]
By definition of infimum and supremum, for every $m\ge N$ we have
\[
u_N \le a_m \le v_N.
\]

Fix $N\in\mathbb{N}$. Because $\{k_n\}$ is strictly increasing, we have $k_n\ge n$ for all $n$,
and in particular there exists $n_0$ such that $k_n\ge N$ for all $n\ge n_0$.
Therefore, for all $n\ge n_0$,
\[
u_N \le a_{k_n} \le v_N.
\]
Now take the limit as $n\to\infty$. Since $u_N$ and $v_N$ are constants (in $n$) and
$a_{k_n}\to L$, we obtain
\[
u_N \le L \le v_N\qquad\text{for every }N\in\mathbb{N}.
\]

Finally, let $N\to\infty$. By definition,
\[
\liminf_{n\to\infty} a_n=\lim_{N\to\infty}u_N,
\qquad
\limsup_{n\to\infty} a_n=\lim_{N\to\infty}v_N.
\]
Taking limits in the inequality $u_N\le L\le v_N$ gives
\[
\liminf_{n\to\infty} a_n \le L \le \limsup_{n\to\infty} a_n,
\]
as desired.
\qed

v_N^{(a)}=\sup\{a_n:n\ge N\}, 
\qquad
v_N^{(b)}=\sup\{b_n:n\ge N\},

and
\[
v_N^{(a+b)}=\sup\{a_n+b_n:n\ge N\}.

For every $n\ge N$, we have

a_n + b_n \le v_N^{(a)} + v_N^{(b)}.

\problem[]{4}
Let $\{a_n\}_{n\ge 1}$ be a sequence. Prove that
$\limsup_{n\to\infty} |a_n| = 0$ if and only if $\lim_{n\to\infty} a_n = 0$.

\begin{scratch}
This is another ``if and only if'' statement, so we prove both directions.

A helpful way to think about $\limsup$ is via 
\[
\limsup_{n\to\infty} x_n \,=\, \lim_{N\to\infty} \Big(\sup\{x_n:n\ge N\}\Big)
\]
whenever the sequence $\{x_n\}$ is bounded above (and if it is not bounded above, the $\limsup$ is $+\infty$).

Here $x_n=|a_n|\ge 0$ always. So:

\begin{itemize}
\item If $a_n\to 0$, then eventually $|a_n|$ is small; consequently every tail supremum $\sup\{|a_n|:n\ge N\}$ must also be small, forcing $\limsup |a_n|=0$.
\item Conversely, if the tail suprema of $|a_n|$ tend to $0$, then in particular each term $|a_n|$ in the tail is bounded by something $<\varepsilon$, which is exactly the definition of $a_n\to 0$.
\end{itemize}
\end{scratch}

\textbf{Proof.}\\
($\Rightarrow$) Assume $\lim_{n\to\infty} a_n=0$. Let $\varepsilon>0$ be given. By definition of convergence, there exists $N_0\in\mathbb{N}$ such that
\[
|a_n-0|=|a_n|<\varepsilon \qquad \text{for all } n\ge N_0.
\]
Fix any $N\ge N_0$. Then for every $n\ge N$ we have $|a_n|<\varepsilon$, and hence
\[
\sup\{|a_n|:n\ge N\}\le \varepsilon.
\]
Define $v_N:=\sup\{|a_n|:n\ge N\}$. The inequality above shows that for all $N\ge N_0$, $v_N\le \varepsilon$.
Taking $N\to\infty$ yields
\[
\limsup_{n\to\infty}|a_n| 
= \lim_{N\to\infty} v_N 
\le \varepsilon.
\]
Because $\varepsilon>0$ was arbitrary and $\limsup_{n\to\infty}|a_n|\ge 0$ (since $|a_n|\ge 0$), we conclude
$\limsup_{n\to\infty}|a_n|=0$.

($\Leftarrow$) Assume $\limsup_{n\to\infty}|a_n|=0$. Set again
\[
v_N:=\sup\{|a_n|:n\ge N\}.
\]
By definition of $\limsup$ (and since $|a_n|\ge 0$ is bounded above whenever $\limsup |a_n|=0$ is finite), we have
$\lim_{N\to\infty} v_N=0$.
Let $\varepsilon>0$ be given. Then there exists $N_0\in\mathbb{N}$ such that
\[
v_{N_0}<\varepsilon.
\]
But for every $n\ge N_0$, we have $|a_n|\le v_{N_0}$ (because $v_{N_0}$ is the supremum of the tail $\{|a_n|:n\ge N_0\}$). Therefore, for all $n\ge N_0$,
\[
|a_n|\le v_{N_0}<\varepsilon.
\]
This is exactly the definition of $a_n\to 0$. Hence $\lim_{n\to\infty} a_n=0$.
\qed

\subsection*{Sequences and Cauchy}
\begin{definition}
We say that $\{a_n\}_{n\ge 1}$ is a Cauchy sequence if:
for all $\varepsilon>0$ there exists $N\in\mathbb{N}$ such that
$|a_n-a_m|<\varepsilon$ for all $m,n\ge N$.
\end{definition}

\begin{theorem}[Cauchy criterion]
Let $\{a_n\}_{n\ge 1}$ be a sequence. The sequence $\{a_n\}_{n\ge 1}$ is
Cauchy if and only if it converges.
\end{theorem}

\begin{proposition}
If $\{a_n\}_{n\ge 1}$ converges, then it is Cauchy.
\end{proposition}

\begin{lemma}
If $\{a_n\}_{n\ge 1}$ is a Cauchy sequence, then it is bounded.
\end{lemma}

\begin{proposition}
If $\{a_n\}_{n\ge 1}$ is a Cauchy sequence, then it converges.
\end{proposition}

\subsection*{Subsequences}
\begin{definition}
Let $\{a_n\}_{n\ge 1}$ be a sequence. We say $\{a_{k_n}\}_{n\ge 1}$ is a
subsequence of $\{a_n\}_{n\ge 1}$ if $\{k_n\}_{n\ge 1}$ is a strictly increasing
sequence of natural numbers: $k_n\in\mathbb{N}$ and $k_n<k_{n+1}$ for all
$n\in\mathbb{N}$.
\end{definition}

\begin{remark}
Any such $\{k_n\}_{n\ge 1}$ must satisfy $k_n\ge n$ for all $n\ge 1$.
This is easy to prove by induction. (Exercise.)
\end{remark}

\begin{theorem}
Let $\{a_n\}_{n\ge 1}$ be a sequence and $a\in\mathbb{R}$. The sequence
$\{a_n\}_{n\ge 1}$ converges to $a$ if and only if every subsequence
$\{a_{k_n}\}_{n\ge 1}$ converges to $a$.
\end{theorem}

\subsection*{Monotone Subsequences}
\begin{proposition}
For any sequence, there is a subsequence that is monotone.
\end{proposition}

\begin{theorem}[Bolzano--Weierstrass]
For any bounded sequence, there exists a subsequence that converges.
\end{theorem}

\subsection*{Divergence to $\pm\infty$}
\begin{definition}
Let $\{a_n\}_{n\ge 1}$ be a sequence. We say:
\begin{enumerate}[label=\textbf{(\arabic*)}]
\item $\{a_n\}_{n\ge 1}$ diverges to $+\infty$ if for all $M>0$ there exists
$N\in\mathbb{N}$ such that $a_n>M$ for all $n\ge N$. When this is true, we write
$\lim_{n\to\infty} a_n=+\infty$.
\item $\{a_n\}_{n\ge 1}$ diverges to $-\infty$ if for all $m<0$ there exists
$N\in\mathbb{N}$ such that $a_n<m$ for all $n\ge N$. When this is true, we write
$\lim_{n\to\infty} a_n=-\infty$.
\end{enumerate}
\end{definition}

\begin{proposition}
If $\{a_n\}_{n\ge 1}$ is unbounded and increasing (decreasing), then
$\{a_n\}_{n\ge 1}$ diverges to $+\infty$ ($-\infty$, respectively).
\end{proposition}

\begin{remark}
If a sequence is increasing, there are only two possibilities:
\begin{enumerate}[label=\textbf{(\arabic*)}]
\item bounded $\Rightarrow$ converges;
\item unbounded $\Rightarrow$ diverges to $+\infty$.
\end{enumerate}
\end{remark}

\begin{remark}
``$\lim a_n$'' only sometimes makes sense:
\begin{enumerate}[label=\textbf{(\arabic*)}]
\item converges $\Rightarrow \lim_{n\to\infty} a_n=a\in\mathbb{R}$;
\item diverges: diverges to $\pm\infty \Rightarrow \lim_{n\to\infty} a_n=\pm\infty$,
else $\lim_{n\to\infty} a_n$ does not exist.
\end{enumerate}
\end{remark}

\subsection*{Limit Superior and Limit Inferior}
\begin{definition}
Let $\{a_n\}_{n\ge 1}$ be a sequence (convergent or not). Case 1:
$\{a_n\}_{n\ge 1}$ is bounded. Fix any $N\in\mathbb{N}$. Then the set
$\{a_n:n\ge N\}$ is nonempty and bounded, so
\[
u_N=\inf\{a_n:n\ge N\},\qquad v_N=\sup\{a_n:n\ge N\}
\]
both exist. We automatically have $u_N\le v_N$ for all $N$.
\end{definition}

\begin{proposition}
With $u_N$ and $v_N$ as above, $\{u_N\}_{N\ge 1}$ is increasing and
$\{v_N\}_{N\ge 1}$ is decreasing. In particular,
\[
u_1\le u_2\le \cdots \le u_N \le v_N \le \cdots \le v_2 \le v_1
\quad\text{for all }N\ge 1.
\]
\end{proposition}

\begin{proposition}
The sequences $\{u_N\}_{N\ge 1}$ and $\{v_N\}_{N\ge 1}$ are both monotone and
bounded, and therefore converge. Set
\[
u=\lim_{N\to\infty} u_N
\qquad\text{and}\qquad
v=\lim_{N\to\infty} v_N.
\]
Then $u\le v$. Moreover, if $\lim_{n\to\infty} a_n=a$ exists, then
$u\le a\le v$.
\end{proposition}

\begin{definition}
Let $\{a_n\}_{n\ge 1}$ be a bounded sequence. We define
\[
\limsup_{n\to\infty} a_n=\lim_{N\to\infty} v_N
=\lim_{N\to\infty}\Big(\sup\{a_n:n\ge N\}\Big)
\]
and
\[
\liminf_{n\to\infty} a_n=\lim_{N\to\infty} u_N
=\lim_{N\to\infty}\Big(\inf\{a_n:n\ge N\}\Big).
\]
\end{definition}

\begin{corollary}
If $\{a_n\}_{n\ge 1}$ is bounded, then
\[
\liminf_{n\to\infty} a_n \le \limsup_{n\to\infty} a_n.
\]
\end{corollary}

\begin{definition}
If $\{a_n\}_{n\ge 1}$ is not bounded above, then we define
\[
\limsup_{n\to\infty} a_n=+\infty.
\]
If $\{a_n\}_{n\ge 1}$ is not bounded below, then we define
\[
\liminf_{n\to\infty} a_n=-\infty.
\]
\end{definition}

\begin{remark}
So $\limsup$ and $\liminf$ make sense for any sequence, but their values are in
$\mathbb{R}\cup\{\pm\infty\}$.
\end{remark}

\begin{corollary}
For any sequence $\{a_n\}_{n\ge 1}$, we have
\[
\liminf_{n\to\infty} a_n \le \limsup_{n\to\infty} a_n.
\]
\end{corollary}

\begin{theorem}
Let $\{a_n\}_{n\ge 1}$ be a sequence.
\begin{enumerate}[label=\textbf{(\arabic*)}]
\item If $\lim_{n\to\infty} a_n$ exists in $\mathbb{R}\cup\{\pm\infty\}$, then
\[
\liminf_{n\to\infty} a_n=\lim_{n\to\infty} a_n=\limsup_{n\to\infty} a_n.
\]
\item If $\liminf_{n\to\infty} a_n=\limsup_{n\to\infty} a_n$ in
$\mathbb{R}\cup\{\pm\infty\}$, then $\lim_{n\to\infty} a_n$ exists and equals
this value.
\end{enumerate}
\end{theorem}

\begin{remark}
Here, ``$\lim_{n\to\infty} a_n$ exists in $\mathbb{R}\cup\{\pm\infty\}$'' just
means ``$\{a_n\}_{n\ge 1}$ converges, or diverges to $+\infty$, or diverges to
$-\infty$.''
\end{remark}
