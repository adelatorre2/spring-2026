\problem[]{1}
Let $(F, +, \cdot)$ be a field.
\begin{enumerate}
\item Prove that the multiplicative identity is unique: If $b \in F$ satisfies $b \cdot a = a \cdot b = a$ for all $a \in F$, then $b = 1$.
\item Prove that multiplicative inverses are unique: If $a \in F \setminus \{0\}$ and $b \in F$ satisfy $b \cdot a = a \cdot b = 1$, then $b = a^{-1}$.
\item Prove that if $a \cdot b = a \cdot c$ and $a \neq 0$, then $b = c$.
\item Prove that if $a, b \in F$ then $-a + (-b) = -(a + b)$.
\item Prove that if $a, b \in F$ and $a, b \neq 0$, then $a^{-1} \cdot b^{-1} = (ab)^{-1}$.
\end{enumerate}

\begin{scratch}
\textbf{Main ideas (and why we cannot “just divide”).}
In a general field, “division” is not a primitive operation. Instead, we justify any division-like step
by introducing multiplicative inverses via \textbf{(M5)} and then using associativity \textbf{(M3)} and the identity \textbf{(M4)}.

\begin{itemize}
\item \textbf{(a)} Compare $b\cdot 1$ computed from the hypothesis and from \textbf{(M4)}.
\item \textbf{(b)} Insert $1=a\cdot a^{-1}$ (from \textbf{(M5)}) and reassociate (by \textbf{(M3)}) to force $b=a^{-1}$.
\item \textbf{(c)} Prove cancellation by multiplying $a\cdot b=a\cdot c$ on the left by $a^{-1}$ (existence from \textbf{(M5)}), then use \textbf{(M3)} and \textbf{(M4)}.
\item \textbf{(d)} Show $(-a)+(-b)$ is an additive inverse of $(a+b)$ by checking $(a+b)+((-a)+(-b))=0$ using \textbf{(A2)}, \textbf{(A3)}, \textbf{(A4)}, \textbf{(A5)}. Conclude it equals $-(a+b)$ by uniqueness of additive inverses (proved from the axioms, as in lecture).
\item \textbf{(e)} Show $a^{-1}\cdot b^{-1}$ is an inverse of $ab$ using commutativity \textbf{(M2)}, associativity \textbf{(M3)}, and \textbf{(M5)}. Then apply (b) to conclude $a^{-1}b^{-1}=(ab)^{-1}$.
\end{itemize}
\end{scratch}

\textbf{Proof.}\\
Let $(F,+,\cdot)$ be a field.

\medskip
\textbf{(a)} Suppose $b\in F$ satisfies
\[
b\cdot a = a\cdot b = a \quad \text{for all } a\in F.
\]
Let $1\in F$ be the multiplicative identity whose existence is guaranteed by \textbf{(M4)}.
Setting $a=1$ in the hypothesis gives
\[
b\cdot 1 = 1.
\]
But by \textbf{(M4)} (applied with $a=b$), we also have $b\cdot 1=b$. Therefore $b=1$, so the
multiplicative identity is unique.

\medskip
\textbf{(b)} Suppose $a\in F\setminus\{0\}$ and $b\in F$ satisfy
\[
b\cdot a = a\cdot b = 1.
\]
By the multiplicative inverse axiom \textbf{(M5)}, there exists $a^{-1}\in F$ such that
\[
a\cdot a^{-1} = a^{-1}\cdot a = 1.
\]
We prove that $b=a^{-1}$. Using the multiplicative identity axiom \textbf{(M4)}, we have
\[
b = b\cdot 1.
\]
Substituting $1=a\cdot a^{-1}$ from \textbf{(M5)} gives
\[
b = b\cdot (a\cdot a^{-1}).
\]
By associativity of multiplication \textbf{(M3)},
\[
b\cdot (a\cdot a^{-1}) = (b\cdot a)\cdot a^{-1}.
\]
Using the hypothesis $b\cdot a=1$, this becomes
\[
(b\cdot a)\cdot a^{-1} = 1\cdot a^{-1}.
\]
Finally, by \textbf{(M4)} we have $1\cdot a^{-1}=a^{-1}$. Therefore $b=a^{-1}$, so the
multiplicative inverse of $a$ is unique.

\medskip
\textbf{(c)} Suppose $a,b,c\in F$ satisfy $a\cdot b=a\cdot c$ and $a\neq 0$. By \textbf{(M5)}, there exists
$a^{-1}\in F$ such that $a^{-1}\cdot a=1$. Multiply the equality $a\cdot b=a\cdot c$ on the left by $a^{-1}$:
\[
a^{-1}\cdot (a\cdot b)=a^{-1}\cdot (a\cdot c).
\]
By associativity \textbf{(M3)},
\[
(a^{-1}\cdot a)\cdot b=(a^{-1}\cdot a)\cdot c.
\]
Using $a^{-1}\cdot a=1$ (from \textbf{(M5)}) and the identity axiom \textbf{(M4)}, we get
\[
1\cdot b=1\cdot c \quad\Rightarrow\quad b=c.
\]

\medskip
\textbf{(d)} We claim that $-a+(-b)$ is the additive inverse of $a+b$, hence equals $-(a+b)$.
First compute, using associativity \textbf{(A3)} and commutativity \textbf{(A2)} of $+$,
\begin{align*}
(a+b)+\bigl((-a)+(-b)\bigr)
&= a+\Bigl(b+\bigl((-a)+(-b)\bigr)\Bigr) \quad \text{by \textbf{(A3)}}\\
&= a+\Bigl(\bigl(b+(-b)\bigr)+(-a)\Bigr) \quad \text{by \textbf{(A3)} and \textbf{(A2)}}\\
&= a+\bigl(0+(-a)\bigr) \quad \text{by \textbf{(A5)}}\\
&= a+(-a) \quad \text{by \textbf{(A4)}}\\
&= 0 \quad \text{by \textbf{(A5)}}.
\end{align*}
Thus $(-a)+(-b)$ is an additive inverse of $a+b$.

\emph{Uniqueness of additive inverses (from axioms).} If $x,y\in F$ both satisfy $(a+b)+x=0$ and $(a+b)+y=0$,
then adding $-(a+b)$ to both sides and using \textbf{(A3)}, \textbf{(A4)}, \textbf{(A5)} yields $x=y$ (as in lecture).
Therefore the additive inverse of $a+b$ is unique, and since $-(a+b)$ denotes that unique inverse, we conclude
\[
-a+(-b)=(-a)+(-b)=-(a+b).
\]

\medskip
\textbf{(e)} Assume $a,b\in F$ with $a\neq 0$ and $b\neq 0$. By \textbf{(M5)}, there exist $a^{-1},b^{-1}\in F$ such that
$a\cdot a^{-1}=1$ and $b\cdot b^{-1}=1$. Consider
\[
(ab)\cdot (a^{-1}\cdot b^{-1}).
\]
By associativity \textbf{(M3)} and commutativity \textbf{(M2)},
\[
(ab)\cdot (a^{-1}b^{-1}) = a\cdot b\cdot a^{-1}\cdot b^{-1}
= (a\cdot a^{-1})\cdot (b\cdot b^{-1}).
\]
Using \textbf{(M5)} and then \textbf{(M4)}, this equals $1\cdot 1=1$. Similarly,
\[
(a^{-1}b^{-1})\cdot (ab)=1.
\]
Hence $a^{-1}b^{-1}$ is a multiplicative inverse of $ab$. By part \textbf{(b)} (uniqueness of inverses),
\[
a^{-1}\cdot b^{-1}=(ab)^{-1}.
\]
\qed

\problem[]{2}
Given two points $(x_1, y_1)$ and $(x_2, y_2)$ in $\mathbb{R}^2 = \{(x,y) : x,y \in \mathbb{R}\}$, we write
\[
(x_1, y_1) < (x_2, y_2) \Leftrightarrow x_1 < x_2 \text{ or } (x_1 = x_2 \text{ and } y_1 < y_2).
\]
\begin{enumerate}
\item Prove that $<$ is an order relation on $\mathbb{R}^2$.
\item Prove that $(\mathbb{R}^2, <)$ does not satisfy the least upper bound property. (Hint: Consider the set of all points on the y-axis.)
\end{enumerate}

\begin{scratch}
\textbf{Main ideas.}
This is the \emph{lexicographic order} on $\mathbb{R}^2$.

\begin{itemize}
\item \textbf{(a)} To prove $<$ is an order relation on $\mathbb{R}^2$, verify the two defining properties:
\textbf{trichotomy} and \textbf{transitivity}. The proofs reduce to the corresponding properties of the usual order on $\mathbb{R}$ by splitting into cases on the first coordinates.
\item \textbf{(b)} To show the least upper bound property fails, use the set
$A=\{(0,y): y\in\mathbb{R}\}$ (the $y$-axis). It is bounded above (e.g. by $(1,0)$), but any upper bound must have first coordinate $>0$, and then you can always find a strictly smaller upper bound by decreasing the first coordinate (e.g. replace $(x,y)$ by $(x/2,0)$). Hence no least upper bound exists.
\end{itemize}
\end{scratch}

\textbf{Proof.}\\
Let $p=(x_1,y_1)$, $q=(x_2,y_2)$, and $r=(x_3,y_3)$ be points in $\mathbb{R}^2$.

\medskip
\textbf{(a)} We prove that $<$ is an order relation on $\mathbb{R}^2$ in the sense of the lecture definition.

\emph{Trichotomy.} By trichotomy of the usual order on $\mathbb{R}$, exactly one of $x_1<x_2$, $x_1=x_2$, or $x_2<x_1$ holds.

\begin{itemize}
\item If $x_1<x_2$, then $p<q$ by definition.
\item If $x_2<x_1$, then $q<p$ by definition.
\item If $x_1=x_2$, then by trichotomy on $\mathbb{R}$ applied to $y_1,y_2$, exactly one of $y_1<y_2$, $y_1=y_2$, or $y_2<y_1$ holds. In the first case, $p<q$; in the second, $p=q$; in the third, $q<p$.
\end{itemize}
Therefore exactly one of $p<q$, $p=q$, or $q<p$ holds.

\emph{Transitivity.} Assume $p<q$ and $q<r$. We show $p<r$.

If $x_1<x_2$, then from $q<r$ we have either $x_2<x_3$ or $x_2=x_3$. In both cases $x_2\le x_3$, hence $x_1<x_3$, so $p<r$ by definition.

If instead $x_1=x_2$, then $p<q$ forces $y_1<y_2$. Now consider $q<r$:
\begin{itemize}
\item If $x_2<x_3$, then $x_1=x_2<x_3$, so $p<r$.
\item If $x_2=x_3$, then $q<r$ forces $y_2<y_3$, and by transitivity in $\mathbb{R}$ we get $y_1<y_3$, so (since $x_1=x_3$) we conclude $p<r$.
\end{itemize}
Thus $<$ is transitive.

Since $<$ satisfies trichotomy and transitivity, it is an order relation on $\mathbb{R}^2$.

\medskip
\textbf{(b)} Let
\[
A:=\{(0,y): y\in\mathbb{R}\}\subseteq\mathbb{R}^2.
\]
Then $A$ is nonempty (for example $(0,0)\in A$). We claim $A$ is bounded above in $(\mathbb{R}^2,<)$. Indeed, $(1,0)$ is an upper bound, because for any $(0,y)\in A$ we have $0<1$, hence $(0,y)<(1,0)$ by the definition of $<$.

We show $A$ has \emph{no} least upper bound. First note that if $(u,v)$ is any upper bound of $A$, then necessarily $u>0$. If $u<0$, then $(0,0)\in A$ would satisfy $(u,v)<(0,0)$, so $(u,v)$ could not be an upper bound. If $u=0$, then being an upper bound would require $y\le v$ for all $y\in\mathbb{R}$, which is impossible.

Now take any upper bound $(u,v)$ of $A$; then $u>0$. Define $w:=(u/2,0)$. For any $(0,y)\in A$ we have $0<u/2$, hence $(0,y)<(u/2,0)=w$, so $w$ is also an upper bound of $A$. But $w<(u,v)$ because $u/2<u$.

Therefore every upper bound of $A$ is strictly larger than some other upper bound of $A$, so no least upper bound exists. Hence $(\mathbb{R}^2,<)$ does not satisfy the least upper bound property.
\qed

\problem[]{3}
Recall that the set of complex numbers $\mathbb{C}$ consists of all numbers of the form $a + ib$ for $a, b \in \mathbb{R}$. We can add and multiply complex numbers as follows:
\[
(a + ib) + (c + id) = (a + c) + i(b + d) \quad \text{and} \quad (a + ib)\cdot(c + id) = (ac - bd) + i(ad + bc).
\]
With these operations, it turns out $(\mathbb{C}, +, \cdot)$ is a field (but you do not need to prove this).
\begin{enumerate}
\item The definition of a field says that for any $a + ib \neq 0 + i0$, there exists $c + id \in \mathbb{C}$ so that $(a + ib)\cdot(c + id) = 1 + i0$. What are $c, d \in \mathbb{R}$ in terms of $a, b \in \mathbb{R}$? Prove your answer.
\item Prove that there is no order relation $<$ on $\mathbb{C}$ that makes $(\mathbb{C}, +, \cdot, <)$ an ordered field. (Hint: Can $i = 0 + i1$ be positive? Negative? Zero?)
\end{enumerate}

\begin{scratch}
\textbf{Main ideas.}

\begin{itemize}
\item \textbf{(a)} To find $(a+ib)^{-1}$, write an unknown inverse as $c+id$ and expand
$(a+ib)(c+id)=(ac-bd)+i(ad+bc)$. Matching real and imaginary parts with $1+i0$ gives a $2\times 2$ linear system.
Solve it to get
\[c=\frac{a}{a^2+b^2},\qquad d=-\frac{b}{a^2+b^2}.
\]
The key point is that $a^2+b^2\neq 0$ whenever $a+ib\neq 0$.

\item \textbf{(b)} In an ordered field, we always have $1>0$ and $x\neq 0 \Rightarrow x^2>0$ (Appendix, \textbf{(P3)}).
But in $\mathbb{C}$ we have $i^2=-1$. Trichotomy forces $i>0$, $i=0$, or $i<0$; each case contradicts $x^2>0$ together with $1>0$.
So no order can make $\mathbb{C}$ into an ordered field.
\end{itemize}
\end{scratch}

\textbf{Proof.}\\
\textbf{(a)} Let $a+ib\in\mathbb{C}$ with $a+ib\neq 0+i0$. We seek $c+id\in\mathbb{C}$ such that
\[(a+ib)(c+id)=1+i0.
\]
Using the given multiplication rule,
\[(a+ib)(c+id)=(ac-bd)+i(ad+bc).
\]
Thus we require
\begin{equation}\label{eq:real-imag-system}
ac-bd=1,\qquad ad+bc=0.
\end{equation}
Multiply the first equation in \eqref{eq:real-imag-system} by $a$ and the second by $b$, then add:
\[a(ac-bd)+b(ad+bc)=a.
\]
The left-hand side simplifies to $(a^2+b^2)c$, so
\[(a^2+b^2)c=a.
\]
Similarly, multiply the second equation by $a$ and the first by $b$, then subtract:
\[a(ad+bc)-b(ac-bd)=0-b,
\]
whose left-hand side simplifies to $(a^2+b^2)d$, so
\[(a^2+b^2)d=-b.
\]
Because $a+ib\neq 0$, we have $(a,b)\neq (0,0)$, hence $a^2+b^2>0$ and in particular $a^2+b^2\neq 0$.
Therefore we may divide by $a^2+b^2$ to obtain
\[c=\frac{a}{a^2+b^2},\qquad d=-\frac{b}{a^2+b^2}.
\]
A direct substitution into \eqref{eq:real-imag-system} (or into $(ac-bd)+i(ad+bc)$) shows that these values indeed yield
$(a+ib)(c+id)=1+i0$. This proves the claimed formula for the multiplicative inverse.

\medskip
\textbf{(b)} Suppose, for contradiction, that there exists an order relation $<$ on $\mathbb{C}$ such that
$(\mathbb{C},+,\cdot,<)$ is an ordered field.
By Appendix \textbf{(P3)}, we have $1>0$ and, for any $x\in\mathbb{C}$, $x\neq 0 \Rightarrow x^2>0$.

By trichotomy, exactly one of $i>0$, $i=0$, or $i<0$ holds.
We treat each case:
\begin{itemize}
\item If $i=0$, then $i^2=0$, contradicting $i^2=-1$.
\item If $i>0$, then $i\neq 0$, so Appendix \textbf{(P3)} implies $i^2>0$. But $i^2=-1$, so $-1>0$, hence $1<0$, contradicting $1>0$.
\item If $i<0$, then $-i>0$ (using Appendix \textbf{(P1)} with $a=i$). Since $-i\neq 0$, Appendix \textbf{(P3)} gives $(-i)^2>0$. But $(-i)^2=i^2=-1$, leading again to $-1>0$ and hence $1<0$, a contradiction.
\end{itemize}
All cases contradict the ordered-field consequences, so no such order relation exists.
\qed

\problem[]{4}
Let $A \subseteq \mathbb{R}$ be a nonempty and bounded set, let $c \in \mathbb{R}$ be a number, and define
\[
B = \{c \cdot a : a \in A\}.
\]
\begin{enumerate}
\item Prove that if $c > 0$, then $\sup B = c \cdot \sup A$.
\item Prove that if $c < 0$, then $\sup B = c \cdot \inf A$.
\end{enumerate}

\begin{scratch}
\textbf{Main idea.} We prove an equality of suprema by showing the candidate value is (i) an upper bound for $B$ and
(ii) the \emph{least} such upper bound. The only subtlety is that we may multiply inequalities by a positive number
without changing the direction (ordered-field axiom \textbf{(O2)}), while multiplying by a negative number reverses the
inequality (Appendix \textbf{(P2)}).

\begin{itemize}
\item If $c>0$, then $a\le \sup A$ implies $ca\le c\sup A$, so $c\sup A$ is an upper bound for $B$. For leastness, any
upper bound $M$ of $B$ yields $ca\le M$ for all $a\in A$, hence $a\le M/c$ after multiplying by $c^{-1}>0$, so $M/c$ is an
upper bound for $A$ and $\sup A\le M/c$.
\item If $c<0$, then $\inf A\le a$ implies $c\inf A\ge ca$, so $c\inf A$ is an upper bound for $B$. For leastness, if
$ca\le M$ for all $a\in A$, multiplying by $c^{-1}<0$ reverses the inequality and gives $a\ge M/c$, so $M/c$ is a lower
bound for $A$ and hence $M/c\le \inf A$.
\end{itemize}
\end{scratch}

\textbf{Proof.}\\
Let $A\subseteq\mathbb{R}$ be nonempty and bounded, let $c\in\mathbb{R}$, and set $B=\{ca:a\in A\}$. We use the
definitions of $\sup$ and $\inf$ from the Appendix.

\medskip
\textbf{(a) Assume $c>0$.} We prove $\sup B=c\,\sup A$.

\emph{Step 1: $c\,\sup A$ is an upper bound for $B$.} Let $b\in B$. Then $b=ca$ for some $a\in A$. Because $\sup A$ is an
upper bound for $A$, we have $a\le \sup A$. Since $c>0$, multiplying by $c$ preserves the inequality (ordered-field axiom
\textbf{(O2)}), so
\[
ca\le c\,\sup A.
\]
Thus $b\le c\,\sup A$, and $c\,\sup A$ is an upper bound for $B$.

\emph{Step 2: $c\,\sup A$ is the least upper bound for $B$.} Let $M\in\mathbb{R}$ be any upper bound for $B$. Then for every
$a\in A$, we have $ca\in B$, hence
\[
ca\le M.
\]
Because $c>0$, we have $c\neq 0$ and $c^{-1}>0$. Multiplying by $c^{-1}$ preserves the inequality (\textbf{(O2)}), giving
\[
a\le c^{-1}M = \frac{M}{c} \quad \text{for all } a\in A.
\]
So $M/c$ is an upper bound for $A$. By definition of $\sup A$, this implies $\sup A\le M/c$. Multiplying by $c>0$ again
preserves the inequality (\textbf{(O2)}), so
\[
c\,\sup A\le M.
\]
Therefore $c\,\sup A$ is the least upper bound of $B$, i.e. $\sup B=c\,\sup A$.

\medskip
\textbf{(b) Assume $c<0$.} We prove $\sup B=c\,\inf A$.

\emph{Step 1: $c\,\inf A$ is an upper bound for $B$.} Let $b\in B$, so $b=ca$ for some $a\in A$. Because $\inf A$ is a
lower bound for $A$, we have $\inf A\le a$. Since $c<0$, multiplying by $c$ reverses the inequality (Appendix
\textbf{(P2)}), hence
\[
c\,\inf A \ge ca.
\]
Equivalently, $ca\le c\,\inf A$, so $b\le c\,\inf A$. Thus $c\,\inf A$ is an upper bound for $B$.

\emph{Step 2: $c\,\inf A$ is the least upper bound for $B$.} Let $M\in\mathbb{R}$ be any upper bound for $B$. Then for every
$a\in A$,
\[
ca\le M.
\]
Because $c<0$, we have $c\neq 0$ and $c^{-1}<0$. Multiplying by $c^{-1}$ reverses the inequality (Appendix \textbf{(P2)}),
so
\[
a\ge c^{-1}M = \frac{M}{c} \quad \text{for all } a\in A.
\]
Thus $M/c$ is a lower bound for $A$. By definition of $\inf A$ (greatest lower bound), we get $M/c\le \inf A$.
Multiplying by $c<0$ reverses the inequality again (Appendix \textbf{(P2)}), yielding
\[
M\ge c\,\inf A.
\]
So every upper bound $M$ of $B$ satisfies $c\,\inf A\le M$, hence $c\,\inf A$ is the least upper bound of $B$. Therefore
$\sup B=c\,\inf A$.
\qed

\problem[]{6}

\subsection*{Field axioms (as used in lecture)}
\begin{enumerate}[label=\textbf{(A\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item \textbf{(Closure for $+$)} If $a,b\in F$, then $a+b\in F$.
\item \textbf{(Commutativity of $+$)} If $a,b\in F$, then $a+b=b+a$.
\item \textbf{(Associativity of $+$)} If $a,b,c\in F$, then $(a+b)+c=a+(b+c)$.
\item \textbf{(Additive identity)} There exists $0\in F$ such that $a+0=0+a=a$ for all $a\in F$.
\item \textbf{(Additive inverses)} For each $a\in F$ there exists $-a\in F$ such that
$a+(-a)=(-a)+a=0$.
\end{enumerate}

\begin{enumerate}[label=\textbf{(M\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item \textbf{(Closure for $\cdot$)} If $a,b\in F$, then $a\cdot b\in F$.
\item \textbf{(Commutativity of $\cdot$)} If $a,b\in F$, then $a\cdot b=b\cdot a$.
\item \textbf{(Associativity of $\cdot$)} If $a,b,c\in F$, then $(a\cdot b)\cdot c=a\cdot(b\cdot c)$.
\item \textbf{(Multiplicative identity)} There exists $1\in F$ such that $a\cdot 1=1\cdot a=a$ for all $a\in F$.
\item \textbf{(Multiplicative inverses)} For each $a\in F\setminus\{0\}$ there exists $a^{-1}\in F$ such that
$a\cdot a^{-1}=a^{-1}\cdot a=1$.
\end{enumerate}

\begin{enumerate}[label=\textbf{(D)}, leftmargin=3em, itemsep=0.25em]
\item \textbf{(Distributivity)} If $a,b,c\in F$, then $(a+b)\cdot c=a\cdot c+b\cdot c$.
\end{enumerate}

\subsection*{Derived properties of fields (proved from the axioms in Lecture 7)} % (Feb 6, 2026)
\begin{enumerate}[label=\textbf{(\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item The additive and multiplicative identities are unique.
\item Additive and multiplicative inverses are unique.
\item If $a,b,c\in F$ satisfy $a+b=a+c$, then $b=c$. If $a,b,c\in F$ satisfy $a\cdot b=a\cdot c$ and $a\neq 0$, then $b=c$.
\item For all $a\in F$, we have $a\cdot 0=0\cdot a=0$.
\item For all $a,b\in F$, we have $(-a)\cdot b=a\cdot(-b)=-(a\cdot b)$.
\item For all $a,b\in F$, we have $(-a)\cdot(-b)=a\cdot b$.
\item If $a\cdot b=0$, then $a=0$ or $b=0$.
\end{enumerate}


\subsection*{Order relations and ordered fields (Lectures 7--8)}

\subsubsection*{Order relations}
\textbf{Definition (Order relation).}
Let $A$ be a nonempty set. An \emph{order relation} (or \emph{inequality}) on $A$ is a relation $<$ on $A$ such that:
\begin{enumerate}[label=\textbf{(OR\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item \textbf{(Trichotomy)} If $a,b\in A$, then exactly one of the following holds: $a<b$, or $a=b$, or $b<a$.
\item \textbf{(Transitivity)} If $a,b,c\in A$ satisfy $a<b$ and $b<c$, then $a<c$.
\end{enumerate}

\textbf{Notation.}
\begin{itemize}[leftmargin=2em, itemsep=0.25em]
\item $a>b$ means $b<a$.
\item $a\le b$ means $(a<b)$ or $(a=b)$.
\item $a\ge b$ means $(b<a)$ or $(a=b)$.
\end{itemize}

\subsubsection*{Ordered fields}
\textbf{Definition (Ordered field).}
Let $(F,+,\cdot)$ be a field. We say that $(F,+,\cdot,<)$ is an \emph{ordered field} if it is equipped with an order relation $<$ on $F$ such that:
\begin{enumerate}[label=\textbf{(O\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item If $a,b,c\in F$ satisfy $a<b$, then $a+c<b+c$.
\item If $a,b,c\in F$ satisfy $a<b$ and $0<c$, then $a\cdot c<b\cdot c$.
\end{enumerate}

\textbf{Useful consequences (proved in Lecture 8).}
Let $(F,+,\cdot,<)$ be an ordered field.
\begin{enumerate}[label=\textbf{(P\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item $a>0\iff -a<0$.
\item If $a,b,c\in F$ satisfy $a<b$ and $c<0$, then $a\cdot c>b\cdot c$.
\item If $a\in F$ satisfies $a\neq 0$, then $a^2=a\cdot a>0$. In particular, $1>0$.
\item If $a,b\in F$ satisfy $0<a<b$, then $0<b^{-1}<a^{-1}$.
\end{enumerate}


\subsection*{Bounds, maxima/minima, and suprema/infima (Lecture 8)}

\subsubsection*{Bounds and extrema}
\textbf{Definition (Bounds; max/min).}
Let $(F,+,\cdot,<)$ be an ordered field, and let $A\subseteq F$ be nonempty.
\begin{enumerate}[label=\textbf{(B\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item $A$ is \emph{bounded above} if there exists $M\in F$ such that $a\le M$ for all $a\in A$. Any such $M$ is an \emph{upper bound} for $A$. If, in addition, $M\in A$, then $M$ is the \emph{maximum} of $A$.
\item $A$ is \emph{bounded below} if there exists $m\in F$ such that $m\le a$ for all $a\in A$. Any such $m$ is a \emph{lower bound} for $A$. If, in addition, $m\in A$, then $m$ is the \emph{minimum} of $A$.
\item $A$ is \emph{bounded} if it is both bounded above and bounded below.
\end{enumerate}

\textbf{Remark (Max/min need not exist; uniqueness).}
Some sets do not have a maximum (or a minimum). If a maximum exists, then it is unique: if $M,M'\in A$ both satisfy
$a\le M$ for all $a\in A$ and $a\le M'$ for all $a\in A$, then $M\le M'$ and $M'\le M$, hence $M=M'$.
(The same statement holds for minima.)

\textbf{Remark.}
Even if a set does not have a maximum, it may still have (at least one) upper bound.

\subsubsection*{Least upper bounds and greatest lower bounds}
\textbf{Definition (Supremum/infimum).}
Let $(F,+,\cdot,<)$ be an ordered field, and let $A\subseteq F$ be nonempty.
\begin{enumerate}[label=\textbf{(S\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item A number $L\in F$ is the \emph{least upper bound} (or \emph{supremum}) of $A$ if:
  \begin{itemize}[leftmargin=2em, itemsep=0.25em]
  \item $L$ is an upper bound for $A$, and
  \item if $M$ is an upper bound for $A$, then $L\le M$.
  \end{itemize}
  When this is true, we write $L=\sup A$.
\item A number $\ell\in F$ is the \emph{greatest lower bound} (or \emph{infimum}) of $A$ if:
  \begin{itemize}[leftmargin=2em, itemsep=0.25em]
  \item $\ell$ is a lower bound for $A$, and
  \item if $m$ is a lower bound for $A$, then $m\le \ell$.
  \end{itemize}
  When this is true, we write $\ell=\inf A$.
\end{enumerate}

\textbf{Lemma (Uniqueness of the supremum).}
If $A\subseteq F$ has a least upper bound, then it is unique.

\textbf{Proof.}
Suppose $L,M\in F$ are both least upper bounds for $A$.
Then $L$ is an upper bound and, since $M$ is a least upper bound, we have $M\le L$.
Similarly, $M$ is an upper bound and, since $L$ is a least upper bound, we have $L\le M$.
Therefore $L=M$.\qed


\subsection*{Bounds, maxima/minima, and suprema/infima (Lecture 8)}

\subsubsection*{Bounds and extrema}
\begin{definition}
Let $(F,+,\cdot,<)$ be an ordered field, and let $A\subseteq F$ be nonempty.
\begin{enumerate}[label=\textbf{(B\arabic*)}, leftmargin=3em, itemsep=0.25em]
\item We say that $A$ is \emph{bounded above} if there exists $M\in F$ such that $a\le M$ for all $a\in A$. Any such $M$ is called an \emph{upper bound} for $A$.
If, in addition, $M\in A$, then $M$ is called the \emph{maximum} of $A$.

If, in addition, $m\in A$, then $m$ is called the \emph{minimum} of $A$.

  \end{itemize}

\begin{lemma}
The least upper bound of a set is unique, if it exists.
\end{lemma}

\begin{proof}
Suppose $A\subseteq F$ and $L,M\in F$ are both least upper bounds for $A$.
Because $L$ is a least upper bound, it is an upper bound; because $M$ is a least upper bound, it is an upper bound.
Since $L$ is the \emph{least} upper bound and $M$ is an upper bound, we have $L\le M$.
Similarly, since $M$ is the \emph{least} upper bound and $L$ is an upper bound, we have $M\le L$.
Therefore $L=M$.
\end{proof}
